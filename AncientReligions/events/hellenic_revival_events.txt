# Make Hellenic Paganism the state religion?
character_event = {
	id = ancrel.2102
	desc = EVTDESCancrel.2102
	picture = GFX_evt_roman_temple
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes
	hide_from = yes

	option = {
		name = EVTOPTAancrel.2102
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.25
				trait = zealous
			}
			modifier = {
				factor = 0.8
				trait = cynical
			}
			modifier = {
				factor = 1.25
				trait = proud
			}
			modifier = {
				factor = 0.8
				trait = humble
			}
			modifier = {
				factor = 0.8
				OR = {
					trait = mastermind_theologian
					trait = scholarly_theologian
					trait = scholar
					trait = mystic
				}
			}
		}
		piety = 50
		if = {
			limit = { NOT = { trait = zealous } }
			random = {
				chance = 50
				add_trait = zealous
			}
		}
		custom_tooltip = {
			text = EVTTOOLTIPAancrel.2102
			hidden_tooltip = {
				any_realm_character  = {
					limit = {
						is_adult = yes
						religion = hellenic_pagan
					}
					opinion = { modifier = opinion_hellenic_state_religion who = ROOT }
				}
				any_realm_character  = {
					limit = {
						is_adult = yes
						NOT = { religion_group = pagan_group }
					}
					opinion = { modifier = opinion_forced_hellenic_state_religion who = ROOT }
				}
				capital_scope = {
					religion = ROOT
				}
				any_courtier = {
					limit = { employer = { character = ROOT } }
					religion = ROOT
				}
				any_vassal = {
					limit = { is_playable = no }
					religion = ROOT
				}
				any_vassal = {
					limit = {
						is_playable = yes
						NOT = { religion = ROOT }
					}
					character_event = { id = ancrel.2103 }
				}
			}
		}
	}

	option = {
		name = EVTOPTBancrel.2102
		ai_chance = {
			factor = 50
		}
		if = {
			limit = { trait = zealous }
			remove_trait = zealous
		}
		hidden_tooltip = {
			any_realm_character  = {
				limit = {
					is_adult = yes
					religion = hellenic_pagan
				}
				opinion = { modifier = opinion_no_hellenic_state_religion who = ROOT }
			}
			any_realm_character  = {
				limit = {
					is_adult = yes
					NOT = { religion = hellenic_pagan }
				}
				opinion = { modifier = opinion_no_forced_hellenic_state_religion who = ROOT }
			}
		}
	}
}
